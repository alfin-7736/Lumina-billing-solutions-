const InvoiceGenerator = ({ isPdfReady, org }) => {
const [orgData, setOrgData] = useState({});
const [items, setItems] = useState([{ id: 1, description: 'Service or
Product Description', qty: 1, rate: 100 }]);
const [clientName, setClientName] = useState('Client Name');
useEffect(() => {
if (!org?.id) return;
const orgRef = doc(db, "organizations", org.id);
const unsubscribe = onSnapshot(orgRef, (doc) => {
if (doc.exists()) setOrgData(doc.data());
});
return unsubscribe;
}, [org]);
const handleItemChange = (id, field, value) => setItems(items.map(item
=> item.id === id ? { ...item, [field]: value } : item));
const addItem = () => setItems([...items, { id: Date.now(),
description: '', qty: 1, rate: 0 }]);
const removeItem = (id) => setItems(items.filter(item => item.id !==
id));
const total = useMemo(() => items.reduce((sum, item) => sum +
(item.qty * item.rate), 0), [items]);
const downloadInvoicePDF = () => {
if (!isPdfReady || !window.jspdf) return;
const { jsPDF } = window.jspdf;
const doc = new jsPDF();
// Add logo if it exists
if (orgData.logoUrl) {
const img = new Image();
img.crossOrigin = "Anonymous"; // Important for CORS
img.src = orgData.logoUrl;
img.onload = () => {
doc.addImage(img, 'PNG', 14, 15, 30, 30); // Adjust
position and size as needed
generatePdfContent(doc);
};
img.onerror = () => generatePdfContent(doc); // Proceed
without logo on error
} else {
generatePdfContent(doc);
}
};
const generatePdfContent = (doc) => {
const invoiceNumber = `INV-${new Date().getFullYear()}-
${Math.floor(1000 + Math.random() * 9000)}`;
doc.setFontSize(20); doc.setFont('helvetica', 'bold');
doc.text("INVOICE", 200, 22, { align: 'right' });
doc.setFontSize(11); doc.setFont('helvetica', 'normal');
doc.text(orgData.name || 'Your Company', 14, 55);
doc.text(orgData.address || 'Your Address', 14, 61);
doc.text(orgData.phone || 'Your Phone', 14, 67);
doc.text("Bill To:", 14, 80); doc.text(clientName, 14, 86);
doc.text(`Invoice #: ${invoiceNumber}`, 200, 55, { align: 'right'
});
doc.text(`Date: ${new Date().toLocaleDateString()}`, 200, 61, {
align: 'right' });
doc.autoTable({
head: [["Description", "Quantity", "Rate", "Amount"]],
body: items.map(item => [item.description, item.qty,
`$${Number(item.rate).toFixed(2)}`, `$${(item.qty *
item.rate).toFixed(2)}`]),
startY: 100, theme: 'striped', headStyles: { fillColor: [22,
160, 133] },
});
const finalY = doc.autoTable.previous.finalY;
doc.setFont('helvetica', 'bold'); doc.setFontSize(12);
doc.text(`Total: $${total.toFixed(2)}`, 200, finalY + 15, { align:
'right' });
doc.save(`invoice_${invoiceNumber}.pdf`);
};

const Cashbook = ({ isPdfReady, org }) => {

"transactions"));
const unsubscribe = onSnapshot(q, (snapshot) => {
const data = snapshot.docs.map(doc => ({ ...doc.data(), id:
doc.id }));

setTransactions(data);
});
return unsubscribe;
}, [org]);

const totalIn = transactions.filter(t => t.type ===
'in').reduce((sum, t) => sum + Number(t.amount), 0);
const totalOut = transactions.filter(t => t.type ===
'out').reduce((sum, t) => sum + Number(t.amount), 0);
return { totalIn, totalOut, balance: totalIn - totalOut };
}, [transactions]);
const handleOpenModal = (t = null) => {
setEditingTransaction(t);
setFormData(t ? { ...t, amount: String(t.amount) } : { date: new
Date().toISOString().split('T')[0], description: '', type: 'in', amount:
'' });
setIsModalOpen(true);
};

e.preventDefault();
const amount = parseFloat(formData.amount);

amount <= 0) return;

const [isPdfReady, setIsPdfReady] = useState(false);
const [showSplash, setShowSplash] = useState(true);
const [user, setUser] = useState(null);
const [currentOrg, setCurrentOrg] = useState(null);
const [authLoading, setAuthLoading] = useState(true);
useEffect(() => {
const timer = setTimeout(() => setShowSplash(false), 2500);
const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
setUser(currentUser);
setAuthLoading(false);
if (!currentUser) setCurrentOrg(null);
});
const jspdfScript = document.createElement('script');
jspdfScript.src =
'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';
jspdfScript.async = true;
jspdfScript.onload = () => { const autotableScript =
document.createElement('script'); autotableScript.src =
'https://cdnjs.cloudflare.com/ajax/libs/jspdf-
autotable/3.8.2/jspdf.plugin.autotable.min.js'; autotableScript.async =
true; autotableScript.onload = () => setIsPdfReady(true);
document.body.appendChild(autotableScript); };
document.body.appendChild(jspdfScript);
return () => { clearTimeout(timer); unsubscribe(); };
}, []);
const handleGoogleSignIn = async () => await signInWithPopup(auth, new
GoogleAuthProvider()).catch(err => console.error("Sign-in error:", err));
const handleSignOut = async () => { await signOut(auth);
setPage('home'); };
const AppContent = () => {
if (authLoading) return null;
if (!user) return <LoginPage onSignIn={handleGoogleSignIn} />;
if (!currentOrg) return <OrgSetup user={user}
onOrgSelected={setCurrentOrg} />;
const renderPage = () => {
switch (page) {
case 'cashbook': return <Cashbook isPdfReady={isPdfReady}
org={currentOrg} />;
case 'invoice': return <InvoiceGenerator
isPdfReady={isPdfReady} org={currentOrg} />;
case 'settings': return <OrgSettings org={currentOrg}
setPage={setPage} />;
default: return <div className="p-8 text-center"><h1
className="text-4xl font-bold">Welcome to {currentOrg.name}</h1><p
className="text-slate-400 mt-2">Select an option from the navigation to
get started.</p></div>;
}
};
return (
<>
<header className="p-4 md:p-6 flex justify-between items-
center border-b border-slate-800">
<div className="flex items-center gap-4"><h1
className="text-xl font-bold text-white cursor-pointer flex items-center
gap-2" onClick={() => setPage('home')}><Briefcase className="h-5 w-5
text-sky-400"/> Lumina</h1><div className="flex items-center gap-2 bg-
slate-900 p-1.5 rounded-lg"><Building className="h-4 w-4 text-slate-500
ml-2"/><span className="text-sm font-medium text-slate-300 mr-
2">{currentOrg.name}</span></div></div>
<nav className="flex items-center gap-2">
<div className="hidden md:flex items-center bg-
slate-900 p-1.5 rounded-lg">
<button onClick={() => setPage('home')}
className={`flex items-center gap-2 px-3 py-2 rounded-md text-sm ${page
=== 'home' ? 'bg-slate-50 text-slate-900' : 'text-slate-400 hover:bg-
slate-800'}`}><Home className="h-4 w-4" />Home</button>
<button onClick={() => setPage('cashbook')}
className={`flex items-center gap-2 px-3 py-2 rounded-md text-sm ${page
=== 'cashbook' ? 'bg-slate-50 text-slate-900' : 'text-slate-400 hover:bg-
slate-800'}`}><BookOpen className="h-4 w-4" />Cashbook</button>
<button onClick={() => setPage('invoice')}
className={`flex items-center gap-2 px-3 py-2 rounded-md text-sm ${page
=== 'invoice' ? 'bg-slate-50 text-slate-900' : 'text-slate-400 hover:bg-
slate-800'}`}><FileText className="h-4 w-4" />Invoice</button>
<button onClick={() => setPage('settings')}
className={`flex items-center gap-2 px-3 py-2 rounded-md text-sm ${page
=== 'settings' ? 'bg-slate-50 text-slate-900' : 'text-slate-400 hover:bg-
slate-800'}`}><Settings className="h-4 w-4" />Settings</button>
</div>
<Button variant="o" size="d"
onClick={handleSignOut}><LogOut className="mr-2 h-4 w-4"/>Sign
Out</Button>
</nav>
</header>
<main>{renderPage()}</main>
<footer className="text-center p-6 text-slate-500 text-sm
border-t border-slate-800 mt-8">Powered by Lumina Web Solutions</footer>
</>
);
};
return (
<div className="bg-slate-950 min-h-screen font-sans text-white">
<style>{`@keyframes fade-in { from { opacity: 0; } to {
opacity: 1; } } .animate-fade-in { animation: fade-in 0.5s ease-out
forwards; } @keyframes splash-out { 0% { opacity: 1; } 90% { opacity: 1;
} 100% { opacity: 0; visibility: hidden; } } .animate-splash-out {
animation: splash-out 2.5s ease-in-out forwards; }`}</style>
{showSplash && <SplashScreen />}
<div className={`transition-opacity duration-500 ${showSplash
? 'opacity-0' : 'opacity-100'}`}>
<div className="container mx-auto"><AppContent /></div>
</div>
</div>
);
}
